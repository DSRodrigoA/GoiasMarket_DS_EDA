# Com a conexo realizada e verificada, podemos dar incio  interao com o SQL Server, realizando as queries necessrias
# para seleo de dados e criao de pipelines com as fontes de dados.

# Para este estudo, utilizaremos a bilbioteca Pandas para realizar as queries, j criando os dataframes. A funo do pandas 
# que ser utilizada para realizar a interface com o SQL, ser o 'read_sql', utilizando as subfunes necessrias. 

# A Query SQL pode ser definida abaixo, relacionando a tabela 'FATO' com as outras dimenses do DW atravs de JOIN's pelas PK
# e SK. A forma com que modela a query no Python segue da mesma maneira que no SQL Server.

# Para facilitar o entendimento da funo, foi modelada ainda no SQL Server uma view com exatamente os dados abaixo. Portanto,
# fazemos um SELECT da view 'PIPELINE_FATO' e definimos o parmetro de conexo 'conn' para executar a funo de query do pandas.

# QUERY SQL / VIEW PIPELINE_FATO:

# SELECT TEMPO.DATA, 
#        FATO.IDNOTA AS INVOICE, 
#        CLIENTE.NOME AS CLIENTE, 
#        PRODUTO.NOME AS PRODUTO, 
#        CAT.NOME AS CATEGORIA, 
#        FORNECEDOR.NOME AS FORNECEDOR, 
#        VENDEDOR.NOME AS VENDEDOR, 
#        FORMAPAG.FORMA AS FORMAPAG, 
#        FATO.QUANTIDADE, 
#        FATO.TOTAL_ITEM, 
#        FATO.CUSTO_TOTAL, 
#        FATO.LUCRO_TOTAL 
# FROM FATO FATO
# INNER JOIN DIM_CLIENTE CLIENTE ON FATO.IDCLIENTE = CLIENTE.IDCLIENTE
# INNER JOIN DIM_PRODUTO PRODUTO ON FATO.IDPRODUTO = PRODUTO.IDPRODUTO
# INNER JOIN DIM_VENDEDOR VENDEDOR ON FATO.IDVENDEDOR = VENDEDOR.IDVENDEDOR
# INNER JOIN CATEGORIA CAT ON PRODUTO.ID_CATEGORIA = CAT.IDCATEGORIA
# INNER JOIN DIM_FORNECEDOR FORNECEDOR ON FATO.IDFORNECEDOR = FORNECEDOR.IDFORNECEDOR
# INNER JOIN DIM_TEMPO TEMPO ON FATO.IDTEMPO = TEMPO.IDSK
# INNER JOIN DIM_FORMA FORMAPAG ON FATO.IDFORMA = FORMAPAG.IDFORMA


df_fato = pd.read_sql_query('SELECT * FROM PIPELINE_FATO',conn)
